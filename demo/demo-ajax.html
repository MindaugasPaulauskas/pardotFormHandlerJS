AJAX Demo: AJAX call

This demo is a showcase of an AJAX functionality.

Copy below code to your console and check the results:

pardotFormHandlerJS.ajax(
    "https://pardothandler.onrender.com/pardot-handler-simulation-with-dynamic-success-and-error-location",
    {
        name: "Mindaugas",
        email: "fake@email.io"
    },
    {
        successMessage: "You have subscribed!",
        timeoutMessage: "Something went wrong. Please try again later."
    }
)
    .onSubmit(function(detail) {
        console.log("ajax: [onSubmit] detail:", detail);
    })
    .onSuccess(function(detail) {
        console.log("ajax: [onSuccess] detail:", detail);
    })
    .onError(function(detail) {
        console.log("ajax: [onError] detail:", detail);
    })
    .onComplete(function(detail) {
        console.log("ajax: [onComplete] detail:", detail);
    });
